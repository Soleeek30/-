def calculate_average(*args):

average = total / count

return average


if __name__ == '__main__':

print(f"Среднее арифметическое: {result}")

import math

def calculate_area(a, b, c):

    area = math.sqrt(s * (s - a) * (s - b) * (s - c))
    return area
